# Factorial problem 
 
using System;

public class HelloWorld
{
    static long fact(int n){
        if(n==0 || n==1) return 1;
        return n*fact(n-1);
    }
    public static void Main(string[] args)
    {
        Console.WriteLine ("Enter a number");
        int num=Convert.ToInt32(Console.ReadLine());
        long res=fact(num);
        Console.WriteLine(res);
    }
}
